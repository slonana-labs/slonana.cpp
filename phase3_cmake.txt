cmake_minimum_required(VERSION 3.16)
project(phase3_multi_master_demo VERSION 1.0.0 LANGUAGES CXX)

# Set C++ standard
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Compiler flags for performance
set(CMAKE_CXX_FLAGS_RELEASE "-O3 -DNDEBUG")

# Include directories
include_directories(${CMAKE_SOURCE_DIR}/include)

# Multi-master component sources only
set(MULTI_MASTER_SOURCES
    src/cluster/multi_master_manager.cpp
    src/cluster/multi_master_coordinator.cpp
    src/network/topology_manager.cpp
    src/network/distributed_load_balancer.cpp
    src/common/types.cpp
    src/phase3_multi_master_demo.cpp
)

# Create executable
add_executable(phase3_multi_master_demo ${MULTI_MASTER_SOURCES})
target_include_directories(phase3_multi_master_demo PUBLIC include)

# Link threading library
find_package(Threads REQUIRED)
target_link_libraries(phase3_multi_master_demo Threads::Threads)
